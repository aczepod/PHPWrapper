 54:     /**
 55:      * <h4>Instantiate class</h4>
 56:      *
 57:      * The method accepts the same parameters as the {@link MongoCollection} constructor,
 58:      * the first one represents the database object, the second represents the container
 59:      * name, if the latter is missing, the method will raise an exception.
 60:      *
 61:      * The method will also accept a database object in the first parameter, in that case
 62:      * the second parameter will be ignored.
 63:      *
 64:      * The method will first attempt to instantiate the {@link MongoCollection} object, it
 65:      * will then store it in the connection data member.
 66:      *
 67:      * @param mixed                 $theConnection      Server or database.
 68:      * @param mixed                 $theOptions         Database name.
 69:      *
 70:      * @access public
 71:      *
 72:      * @uses Connection()
 73:      */
 74:     public function __construct( $theConnection = NULL, $theOptions = NULL )
 75:     {
 76:         //
 77:         // Check server.
 78:         //
 79:         if( $theConnection !== NULL )
 80:         {
 81:             //
 82:             // Handle container.
 83:             //
 84:             if( $theConnection instanceof self )
 85:                 parent::__construct( $theConnection->Connection() );
 86:             
 87:             //
 88:             // Handle Mongo collection.
 89:             //
 90:             elseif( $theConnection instanceof MongoCollection )
 91:                 parent::__construct( $theConnection );
 92:             
 93:             //
 94:             // Handle database.
 95:             //
 96:             else
 97:             {
 98:                 //
 99:                 // Check container name.
100:                 //
101:                 if( $theOptions === NULL )
102:                     throw new Exception
103:                         ( "Missing container name",
104:                           kERROR_MISSING );                                     // !@! ==>
105:                 
106:                 //
107:                 // Handle database class.
108:                 //
109:                 if( $theConnection instanceof CMongoDatabase )
110:                     $theConnection = $theConnection->Connection();
111:                 
112:                 //
113:                 // Check Mongo database class.
114:                 //
115:                 if( $theConnection instanceof MongoDB )
116:                     parent::__construct
117:                         ( $theConnection->selectCollection( (string) $theOptions ) );
118:                 
119:                 else
120:                     throw new Exception
121:                         ( "Invalid database connection type",
122:                           kERROR_PARAMETER );                                   // !@! ==>
123:             
124:             } // Provided database.
125:         
126:         } // Provided parameter.
127:         
128:         //
129:         // Call parent constructor, just in case.
130:         //
131:         else
132:             parent::__construct();
133:         
134:     } // Constructor.

179:     /**
180:      * <h4>Manage native connection</h4>
181:      *
182:      * We overload this method to ensure that the provided container is a
183:      * {@link MongoCollection} object.
184:      *
185:      * @param mixed                 $theValue           Persistent container or operation.
186:      * @param boolean               $getOld             <tt>TRUE</tt> get old value.
187:      *
188:      * @access public
189:      * @return mixed                <i>New</i> or <i>old</i> native container.
190:      *
191:      * @throws Exception
192:      */
193:     public function Connection( $theValue = NULL, $getOld = FALSE )
194:     {
195:         //
196:         // Check new value.
197:         //
198:         if( ($theValue !== NULL)
199:          && ($theValue !== FALSE) )
200:         {
201:             //
202:             // Check value type.
203:             //
204:             if( ! $theValue instanceof MongoCollection )
205:                 throw new Exception
206:                     ( "Invalid container type",
207:                       kERROR_PARAMETER );                                       // !@! ==>
208:         
209:         } // New value.
210:         
211:         //
212:         // Set connection.
213:         //
214:         $save = parent::Connection( $theValue, $getOld );
215:         
216:         //
217:         // Handle changes.
218:         //
219:         if( $theValue !== NULL )
220:         {
221:             //
222:             // Handle new value.
223:             //
224:             if( $theValue !== FALSE )
225:                 $this->offsetSet( kOFFSET_NAME, $theValue->getName() );
226:             
227:             //
228:             // Remove value.
229:             //
230:             else
231:                 $this->exchangeArray( Array() );
232:         
233:         } // Made changes.
234:         
235:         return $save;                                                               // ==>
236: 
237:     } // Connection.

253:     /**
254:      * <h4>Manage an object from the container</h4>
255:      *
256:      * We implement this method to handle MongoCollection object stores.
257:      *
258:      * For more information see {@link CContainer::ManageObject()}.
259:      *
260:      * <i>Note: the commit operations are performed by default with the <tt>safe</tt>
261:      * option.</i>
262:      *
263:      * @param reference            &$theObject          Object.
264:      * @param mixed                 $theIdentifier      Identifier.
265:      * @param mixed                 $theModifiers       Options or offsets list.
266:      *
267:      * @access public
268:      * @return mixed                The native operation status.
269:      *
270:      * @throws Exception
271:      *
272:      * @uses Connection()
273:      *
274:      * @see kFLAG_PERSIST_MASK kFLAG_PERSIST_INSERT kFLAG_PERSIST_UPDATE
275:      * @see kFLAG_PERSIST_REPLACE kFLAG_PERSIST_MODIFY kFLAG_PERSIST_DELETE
276:      */
277:     public function ManageObject( &$theObject,
278:                                    $theIdentifier = NULL,
279:                                    $theModifiers = kFLAG_DEFAULT )
280:     {
281:         //
282:         // Get container.
283:         //
284:         $container = $this->Connection();
285:         if( ! ($container instanceof MongoCollection) )
286:             throw new Exception
287:                 ( "Missing or invalid native container",
288:                   kERROR_STATE );                                               // !@! ==>
289:         
290:         //
291:         // Handle retrieve.
292:         //
293:         if( is_array( $theModifiers )                   // Offsets list,
294:          || (! ($theModifiers & kFLAG_PERSIST_MASK)) )  // or retrieve operation.
295:         {
296:             //
297:             // Determine criteria.
298:             //
299:             if( $theIdentifier !== NULL )
300:                 $criteria = array( kOFFSET_NID => $theIdentifier );
301:             elseif( is_array( $theObject )
302:                  && array_key_exists( kOFFSET_NID, $theObject ) )
303:                 $criteria = array( kOFFSET_NID => $theObject[ kOFFSET_NID ] );
304:             elseif( ($theObject instanceof ArrayObject)
305:                  && $theObject->offsetExists( kOFFSET_NID ) )
306:                 $criteria = array( kOFFSET_NID => $theObject->offsetGet( kOFFSET_NID ) );
307:             else
308:                 throw new Exception
309:                     ( "Missing object identifier",
310:                       kERROR_STATE );                                           // !@! ==>
311:             
312:             //
313:             // Determine selected offsets.
314:             //
315:             $fields = Array();
316:             if( is_array( $theModifiers )
317:              && count( $theModifiers ) )
318:             {
319:                 foreach( $theModifiers as $offset )
320:                     $fields[ $offset ] = TRUE;
321:             }
322:             
323:             //
324:             // Locate object.
325:             //
326:             $theObject = $container->findOne( $criteria, $fields );
327:             
328:             return ( $theObject !== NULL );                                         // ==>
329:         
330:         } // Retrieve
331:         
332:         //
333:         // Get options mask.
334:         //
335:         $modifiers = $theModifiers & kFLAG_PERSIST_MASK;
336:         
337:         //
338:         // Init commit options.
339:         //
340:         $options = array( 'safe' => TRUE );
341:         
342:         //
343:         // Handle insert.
344:         //
345:         if( $modifiers == kFLAG_PERSIST_INSERT )
346:         {
347:             //
348:             // Set identifier.
349:             //
350:             if( $theIdentifier !== NULL )
351:                 $theObject[ kOFFSET_NID ] = $theIdentifier;
352:             
353:             //
354:             // Convert to object.
355:             // If you provide an array, the missing identifier will not be set by Mongo.
356:             //
357:             if( is_array( $theObject ) )
358:             {
359:                 //
360:                 // Convert.
361:                 //
362:                 $object = new ArrayObject( $theObject );
363:                 
364:                 //
365:                 // Insert.
366:                 //
367:                 $status = $container->insert( $object, $options );
368:                 
369:                 //
370:                 // Restore.
371:                 //
372:                 $theObject = $object->getArrayCopy();
373:             }
374:             else
375:                 $status = $container->insert( $theObject, $options );
376:             
377:             //
378:             // Check status.
379:             //
380:             if( ! $status[ 'ok' ] )
381:                 throw new Exception
382:                     ( $status[ 'errmsg' ],
383:                       kERROR_COMMIT );                                          // !@! ==>
384:             
385:             return $theObject[ kOFFSET_NID ];                                       // ==>
386:         
387:         } // Insert.
388:         
389:         //
390:         // Handle update.
391:         //
392:         if( $modifiers == kFLAG_PERSIST_UPDATE )
393:         {
394:             //
395:             // Set default commit options.
396:             //
397:             $options[ 'upsert' ] = FALSE;       // Don't create new objects.
398:             $options[ 'multiple' ] = FALSE;     // Operate on first object.
399:             
400:             //
401:             // Determine criteria.
402:             //
403:             if( $theIdentifier !== NULL )
404:                 $criteria = array( kOFFSET_NID => $theIdentifier );
405:             elseif( is_array( $theObject )
406:                  && array_key_exists( kOFFSET_NID, $theObject ) )
407:                 $criteria = array( kOFFSET_NID => $theObject[ kOFFSET_NID ] );
408:             elseif( ($theObject instanceof ArrayObject)
409:                  && $theObject->offsetExists( kOFFSET_NID ) )
410:                 $criteria = array( kOFFSET_NID => $theObject->offsetGet( kOFFSET_NID ) );
411:             else
412:                 throw new Exception
413:                     ( "Missing object identifier",
414:                       kERROR_STATE );                                           // !@! ==>
415:             
416:             //
417:             // Update.
418:             //
419:             $status = $container->update( $criteria, $theObject, $options );
420:             
421:             //
422:             // Check status.
423:             //
424:             if( ! $status[ 'ok' ] )
425:                 throw new Exception
426:                     ( $status[ 'errmsg' ],
427:                       kERROR_COMMIT );                                          // !@! ==>
428:             
429:             return $status[ 'updatedExisting' ];                                    // ==>
430:         
431:         } // Update.
432:         
433:         //
434:         // Handle modify.
435:         //
436:         if( $modifiers == kFLAG_PERSIST_MODIFY )
437:         {
438:             //
439:             // Set default commit options.
440:             //
441:             $options[ 'upsert' ] = FALSE;       // Don't create new objects.
442:             $options[ 'multiple' ] = FALSE;     // Operate on first object.
443:             
444:             //
445:             // Determine criteria.
446:             //
447:             if( $theIdentifier !== NULL )
448:                 $criteria = array( kOFFSET_NID => $theIdentifier );
449:             else
450:                 throw new Exception
451:                     ( "Missing object identifier",
452:                       kERROR_STATE );                                           // !@! ==>
453:             
454:             //
455:             // Init modification matrix.
456:             //
457:             $modifications = Array();
458:             
459:             //
460:             // Handle increment.
461:             //
462:             if( ($theModifiers & kFLAG_MODIFY_MASK) == kFLAG_MODIFY_INCREMENT )
463:             {
464:                 //
465:                 // Iterate offsets.
466:                 //
467:                 foreach( $theObject as $offset => $value )
468:                 {
469:                     //
470:                     // Consider only integers.
471:                     //
472:                     $inc = (int) $value;
473:                     
474:                     //
475:                     // Build clause.
476:                     //
477:                     $modifications[ '$inc' ][ $offset ] = $inc;
478:                 }
479:             
480:             } // Increment.
481:             
482:             //
483:             // Handle append.
484:             //
485:             elseif( ($theModifiers & kFLAG_MODIFY_MASK) == kFLAG_MODIFY_APPEND )
486:             {
487:                 //
488:                 // Iterate offsets.
489:                 //
490:                 foreach( $theObject as $offset => $value )
491:                 {
492:                     //
493:                     // Convert array objects.
494:                     //
495:                     if( $value instanceof ArrayObject )
496:                         $value = $value->getArrayCopy();
497:                         
498:                     //
499:                     // Handle array value.
500:                     //
501:                     if( is_array( $value ) )
502:                         $modifications[ '$pushAll' ][ $offset ] = $value;
503:                     
504:                     //
505:                     // Handle scalar.
506:                     //
507:                     else
508:                         $modifications[ '$push' ][ $offset ] = $value;
509:                 }
510:             
511:             } // Append.
512:             
513:             //
514:             // Handle add to set.
515:             //
516:             elseif( ($theModifiers & kFLAG_MODIFY_MASK) == kFLAG_MODIFY_ADDSET )
517:             {
518:                 //
519:                 // Iterate offsets.
520:                 //
521:                 foreach( $theObject as $offset => $value )
522:                 {
523:                     //
524:                     // Convert array objects.
525:                     //
526:                     if( $value instanceof ArrayObject )
527:                         $value = $value->getArrayCopy();
528:                         
529:                     //
530:                     // Handle array value.
531:                     //
532:                     if( is_array( $value ) )
533:                         $modifications[ '$addToSet' ][ $offset ][ '$each' ] = $value;
534:                     
535:                     //
536:                     // Handle scalar.
537:                     //
538:                     else
539:                         $modifications[ '$addToSet' ][ $offset ] = $value;
540:                 }
541:             
542:             } // Add to set.
543:             
544:             //
545:             // Handle pop.
546:             //
547:             elseif( ($theModifiers & kFLAG_MODIFY_MASK) == kFLAG_MODIFY_POP )
548:             {
549:                 //
550:                 // Iterate offsets.
551:                 //
552:                 foreach( $theObject as $offset => $value )
553:                 {
554:                     //
555:                     // Consider only integers.
556:                     //
557:                     $inc = ( ((int) $value) > 0 ) ? 1 : -1;
558:                     
559:                     //
560:                     // Pop.
561:                     //
562:                     $modifications[ '$pop' ][ $offset ] = $inc;
563:                 }
564:             
565:             } // Pop.
566:             
567:             //
568:             // Handle pull.
569:             //
570:             elseif( ($theModifiers & kFLAG_MODIFY_MASK) == kFLAG_MODIFY_PULL )
571:             {
572:                 //
573:                 // Iterate offsets.
574:                 //
575:                 foreach( $theObject as $offset => $value )
576:                 {
577:                     //
578:                     // Convert array objects.
579:                     //
580:                     if( $value instanceof ArrayObject )
581:                         $value = $value->getArrayCopy();
582:                         
583:                     //
584:                     // Handle array value.
585:                     //
586:                     if( is_array( $value ) )
587:                         $modifications[ '$pullAll' ][ $offset ] = $value;
588:                     
589:                     //
590:                     // Handle scalar.
591:                     //
592:                     else
593:                         $modifications[ '$pull' ][ $offset ] = $value;
594:                 }
595:             
596:             } // Pull.
597:             
598:             //
599:             // Handle set & unset.
600:             //
601:             else
602:             {
603:                 //
604:                 // Iterate offsets.
605:                 //
606:                 foreach( $theObject as $offset => $value )
607:                 {
608:                     //
609:                     // Set offset.
610:                     //
611:                     if( $value !== NULL )
612:                     {
613:                         //
614:                         // Convert array objects.
615:                         //
616:                         if( $value instanceof ArrayObject )
617:                             $value = $value->getArrayCopy();
618:                         
619:                         //
620:                         // Set.
621:                         //
622:                         $modifications[ '$set' ][ $offset ] = $value;
623:                     }
624:                     
625:                     //
626:                     // Unset offset.
627:                     //
628:                     else
629:                         $modifications[ '$unset' ][ $offset ] = 1;
630:                 }
631:             
632:             } // Set & unset.
633:             
634:             //
635:             // Update.
636:             //
637:             $status = $container->update( $criteria, $modifications, $options );
638:             
639:             //
640:             // Check status.
641:             //
642:             if( ! $status[ 'ok' ] )
643:                 throw new Exception
644:                     ( $status[ 'errmsg' ],
645:                       kERROR_COMMIT );                                          // !@! ==>
646:             
647:             //
648:             // Load modified object.
649:             //
650:             $theObject = $container->findOne( $criteria );
651:             
652:             return $status[ 'updatedExisting' ];                                    // ==>
653:         
654:         } // Modify.
655:         
656:         //
657:         // Handle replace.
658:         //
659:         if( $modifiers == kFLAG_PERSIST_REPLACE )
660:         {
661:             //
662:             // Set identifier.
663:             //
664:             if( $theIdentifier !== NULL )
665:                 $theObject[ kOFFSET_NID ] = $theIdentifier;
666:             
667:             //
668:             // Convert to object.
669:             // If you provide an array, the missing identifier will not be set by Mongo.
670:             //
671:             if( is_array( $theObject ) )
672:             {
673:                 //
674:                 // Convert.
675:                 //
676:                 $object = new ArrayObject( $theObject );
677:                 
678:                 //
679:                 // Replace.
680:                 //
681:                 $status = $container->save( $object, $options );
682:                 
683:                 //
684:                 // Restore.
685:                 //
686:                 $theObject = $object->getArrayCopy();
687:             }
688:             else
689:                 $status = $container->save( $theObject, $options );
690:             
691:             //
692:             // Check status.
693:             //
694:             if( ! $status[ 'ok' ] )
695:                 throw new Exception
696:                     ( $status[ 'errmsg' ],
697:                       kERROR_COMMIT );                                          // !@! ==>
698:             
699:             return $theObject[ kOFFSET_NID ];                                       // ==>
700:         
701:         } // Replace.
702:         
703:         //
704:         // Handle delete.
705:         //
706:         if( $modifiers == kFLAG_PERSIST_DELETE )
707:         {
708:             //
709:             // Determine criteria.
710:             //
711:             if( $theIdentifier !== NULL )
712:                 $criteria = array( kOFFSET_NID => $theIdentifier );
713:             elseif( is_array( $theObject )
714:                  && array_key_exists( kOFFSET_NID, $theObject ) )
715:                 $criteria = array( kOFFSET_NID => $theObject[ kOFFSET_NID ] );
716:             elseif( ($theObject instanceof ArrayObject)
717:                  && $theObject->offsetExists( kOFFSET_NID ) )
718:                 $criteria = array( kOFFSET_NID => $theObject->offsetGet( kOFFSET_NID ) );
719:             else
720:                 throw new Exception
721:                     ( "Missing object identifier",
722:                       kERROR_STATE );                                           // !@! ==>
723:             
724:             //
725:             // Set options.
726:             //
727:             $options[ 'justOne' ] = TRUE;
728:             
729:             //

731:             //

733:             if( ! $status[ 'ok' ] )
734:                 throw new Exception
735:                     ( $status[ 'errmsg' ],
736:                       kERROR_COMMIT );                                          // !@! ==>
737:             
738:             return $status[ 'n' ];                                                  // ==>
739:         
740:         } // Delete.

742:         throw new Exception

745:         
746:     } // ManageObject.

752: 
753:     /**
754:      * <h4>Check if object exists in container</h4>
755:      *
756:      * In this class we return the number of found objects.
757:      *
758:      * @param mixed                 $theIdentifier      Identifier.
759:      * @param string                $theOffset          Offset.
760:      *
761:      * @access public
762:      * @return boolean              <tt>TRUE</tt> exists.
763:      *
764:      * @throws Exception
765:      *
766:      * @uses Connection()
767:      */
768:     public function CheckObject( $theIdentifier, $theOffset = NULL )
769:     {
770:         //
771:         // Get container.
772:         //
773:         $container = $this->Connection();
774:         if( ! ($container instanceof MongoCollection) )
775:             throw new Exception
776:                 ( "Missing or invalid native container",
777:                   kERROR_STATE );                                               // !@! ==>
778:         
779:         //
780:         // Set default offset.
781:         //
782:         if( $theOffset === NULL )
783:             $theOffset = '_id';
784:         
785:         //
786:         // Set criteria.
787:         //
788:         $fields = array( '_id' => TRUE );
789:         $criteria = array( $theOffset => $theIdentifier );
790:         
791:         //
