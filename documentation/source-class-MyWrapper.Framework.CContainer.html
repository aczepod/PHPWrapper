 29: /**
 30:  * <h3>Persistent objects data store ancestor</h3>
 31:  *
 32:  * This <i>abstract</i> class is the ancestor of all container classes in this library, it
 33:  * implements the interface and workflow that all concrete derived classes should implement.
 34:  *
 35:  * A container object serves the purpose of storing and retrieving objects, in the same way
 36:  * as a database table. The object holds a data member that represents a native container
 37:  * from a database.
 38:  *
 39:  * The public interface declares the following main operations:
 40:  *
 41:  * <ul>
 42:  *  <li>{@link ManageObject()}: This method will <i>insert</i>, <i>update</i>,
 43:  *      <i>modify</i>, <i>delete</i> and <i>load</i> an object from the current container.
 44:  *      This method operates one object at the time and it is mainly used by persistent
 45:  *      objects for persisting in containers.
 46:  *  <li>{@link ConvertBinary()}: This method will convert a binary string to a format
 47:  *      compatible with the current data store.
 48:  *      container.
 49:  * </ul>
 50:  *
 51:  * The class does not implement a concrete data store, derived classes must implement
 52:  * specific storage actions.
 53:  *
 54:  *  @package    MyWrapper
 55:  *  @subpackage Framework
 56:  */

 73:     /**
 74:      * <h4>Manage an object from the container</h4>
 75:      *
 76:      * This method can be used to manage a single object in the container, it can
 77:      * <i>insert</i>, <i>update</i>, <i>replace</i>, <i>modify</i>, <i>retrieve</i> and
 78:      * <i>delete</i> an object from the current container by providing the object and/or
 79:      * its native unique identifier ({@link kOFFSET_NID}).
 80:      *
 81:      * This method expects three parameters:
 82:      *
 83:      * <ul>
 84:      *  <li><tt>$theObject</tt>: The object or the data to be modified.
 85:      *  <li><tt>$theIdentifier</tt>: The native unique identifier ({@link kOFFSET_NID}) of
 86:      *      the object. If the value is <i>NULL</i>, it means that it is the duty of the
 87:      *      current container to set the value, this will generally be the case when
 88:      *      inserting objects; in all other cases the parameter is required.
 89:      *  <li><tt>$theModifiers</tt>: This parameter represents the operation options, it is a
 90:      *      bitfield where the following values apply:
 91:      *   <ul>
 92:      *      <li>{@link kFLAG_PERSIST_INSERT}: The provided object will be inserted in the
 93:      *          container. It is assumed that no other element in the container shares
 94:      *          the same identifier, if not, the method should raise an exception.
 95:      *          In this case the <tt>$theObject</tt> represents the full object and
 96:      *          <tt>$theIdentifier</tt> represents the object unique identifier, or
 97:      *          <tt>NULL</tt>, if it is the duty of the container to create the identifier.
 98:      *          If the identifier was created by the container, it is the duty of this
 99:      *          method to set it into the provided object.
100:      *          In this case the method should return the inserted object's native
101:      *          identifier ({@link kOFFSET_NID}).
102:      *      <li>{@link kFLAG_PERSIST_UPDATE}: The provided object will replace an
103:      *          object existing in the container.
104:      *          In this case the <tt>$theObject</tt> represents the full object and
105:      *          <tt>$theIdentifier</tt> represents the object unique identifier.
106:      *          In this case the method should return <tt>TRUE</tt> if the object was
107:      *          updated and <tt>FALSE</tt> if the object was not found.
108:      *      <li>{@link kFLAG_PERSIST_REPLACE}: The provided object will be inserted
109:      *          if the identifier is missing or it doesn't match any entry in the container,
110:      *          or it will update the object corresponding to the provided identifier. This
111:      *          option represents the combination of the {@link kFLAG_PERSIST_INSERT} and
112:      *          {@link kFLAG_PERSIST_UPDATE} operations.
113:      *          In this case the <tt>$theObject</tt> represents the full object and
114:      *          <tt>$theIdentifier</tt> represents the object unique identifier, or
115:      *          <tt>NULL</tt>, if it is the duty of the container to create the identifier.
116:      *          If the identifier was created by the container, it is the duty of this
117:      *          method to set it into the provided object.
118:      *          In this case the method should return the inserted object's native
119:      *          identifier ({@link kOFFSET_NID}).
120:      *      <li>{@link kFLAG_PERSIST_MODIFY}: This option can be used to apply modifications
121:      *          to a subset of the object.
122:      *          In this case the parameters of this method have the following functions:
123:      *       <ul>
124:      *          <li><tt>$theObject</tt>: This parameter will be an array of elements whose
125:      *              <i>key</i> corresponds to the offset to be considered and the
126:      *              <i>value</i> to the value that the operation will use.
127:      *          <li><tt>$theIdentifier</tt>: This parameter keeps its original meaning, it
128:      *              should contain the native identifier of the object to be modified.
129:      *          <li><tt>$theModifiers</tt>: Another section of this bitfield will hold the
130:      *              flags that determine what specific modification should be performed:
131:      *           <ul>
132:      *              <li>{@link kFLAG_MODIFY_INCREMENT}: This flag indicates an increment or
133:      *                  decrement operation, the method will take the value corresponding
134:      *                  to the offset in <tt>$theObject</tt> parameter and add it to the
135:      *                  object identified by <tt>$theIdentifier</tt> in the container.
136:      *                  For instance, to increment by one the '<tt>A</tt>' offset of object
137:      *                  <tt>X</tt>, you would pass the <tt>array( 'A' => 1 )</tt> in
138:      *                  <tt>$theObject</tt>, <tt>X</tt> in <tt>$theIdentifier</tt> and set
139:      *                  the kFLAG_MODIFY_INCREMENT flag; to decrement by two you would
140:      *                  pass <tt>array( 'A' => -2 )</tt> in <tt>$theObject</tt>, with all
141:      *                  other parameters unchanged.
142:      *                  If the offset does not exist, it will be initialised with the
143:      *                  increment value.
144:      *              <li>{@link kFLAG_MODIFY_APPEND}: This flag indicates that we want to
145:      *                  append a value to an existing array, the method will take the value
146:      *                  corresponding to the offset in <tt>$theObject</tt> and append it to
147:      *                  the offset in <tt>$theObject</tt> for the object identified by
148:      *                  <tt>$theIdentifier</tt> in the container.
149:      *                  For instance, to append the value '<tt>A</tt>' to the array in
150:      *                  offset '<tt>FIELD</tt>' for object <tt>X</tt>, you would pass the
151:      *                  <tt>array( 'FIELD' => 'A' )</tt> in <tt>$theObject</tt>,
152:      *                  <tt>X</tt> in <tt>$theIdentifier</tt> and set the
153:      *                  kFLAG_MODIFY_APPEND flag.
154:      *                  If the field does not exist, it will be created with an array
155:      *                  composed of the provided append value; if the field exists and its
156:      *                  value is not an array, an exception should be raised.
157:      *              <li>{@link kFLAG_MODIFY_ADDSET}: This flag indicates that we want to
158:      *                  add a value to a set, this operation is equivalent to
159:      *                  {@link kFLAG_MODIFY_APPEND}, except that the value will be appended
160:      *                  only if it doesn't exist already in the receiving array.
161:      *              <li>{@link kFLAG_MODIFY_POP}: This flag indicates that we want to remove
162:      *                  the first or last element from an array. The method will take the
163:      *                  value corresponding to the offset in <tt>$theObject</tt> and check
164:      *                  its sign: if positive, the method will remove the first element; if
165:      *                  negative it will remove the last.
166:      *                  For instance, to remove the first element in the '<tt>FIELD</tt>'
167:      *                  offset of object <tt>X</tt>, you would pass the
168:      *                  <tt>array( 'FIELD' => 1 )</tt> in <tt>$theObject</tt>, <tt>X</tt> in
169:      *                  <tt>$theIdentifier</tt> and set the kFLAG_MODIFY_POP flag in
170:      *                  <tt>$theModifiers</tt>; to remove the last element you would pass
171:      *                  <tt>array( 'FIELD' => -1 )</tt> in <tt>$theObject</tt>.
172:      *                  If the offset does not exist, the method should not fail; if the
173:      *                  offset is not an array, the method should raise an exception.
174:      *              <li>{@link kFLAG_MODIFY_PULL}: This flag indicates that we want to
175:      *                  remove all occurrences of a value from an array.
176:      *                  For instance, to remove all occurrances of '<tt>A</tt>' from the
177:      *                  array contained in the '<tt>FIELD</tt>' offset of object <tt>X</tt>, 
178:      *                  you would pass <tt>array( 'FIELD' => 'A' )</tt> in
179:      *                  <tt>$theObject</tt>, <tt>X</tt> in <tt>$theIdentifier</tt> and set
180:      *                  the kFLAG_MODIFY_PULL flag in <tt>$theModifiers</tt>.
181:      *                  If the field exists and its value is not an array, an exception
182:      *                  should be raised.
183:      *              <li>{@link kFLAG_MODIFY_MASK} <i>off</i>: If none of the above flags are
184:      *                  set, it means that the key/value pairs set in <tt>$theObject</tt>
185:      *                  represent offsets to be added or removed, depending on the value
186:      *                  part of the pair: if the value is <tt>NULL</tt>, it means that you
187:      *                  want to remove the offset; any other value will replace the existing
188:      *                  offset or be added to the object.
189:      *           </ul>
190:      *       </ul>
191:      *          In this case the method will return the modified object in the
192:      *          <tt>$theObject</tt> parameter, or raise an exception if the operation fails.
193:      *      <li>{@link kFLAG_PERSIST_DELETE}: This option assumes you want to remove the
194:      *          object from the container.
195:      *          In this case the <tt>$theObject</tt> represents the full object and
196:      *          <tt>$theIdentifier</tt> represents the object unique identifier, what
197:      *          counts is that the object's unique identifier ({@link kOFFSET_NID}) is
198:      *          provided. If the object is not found in the container, the method should
199:      *          not fail.
200:      *          In this case the method should return <tt>TRUE</tt> if the object was
201:      *          deleted and <tt>FALSE</tt> if the object was not found.
202:      *   </ul>
203:      *      If none of the above flags are set, it means that the caller wants to retrieve
204:      *      the object identified by the {@link kOFFSET_NID} offset from the provided object
205:      *      object or from the provided identifier. If found, the provided object will
206:      *      receive the located object and the method will return <tt>TRUE</tt>; if not
207:      *      found, the method will set the provided object to <tt>NULL</tt> and return
208:      *      <tt>FALSE</tt>.
209:      * </ul>
210:      *
211:      * The method should raise an exception if the operation was not successful and update
212:      * the provided object with whatever data the operation will generate.
213:      *
214:      * The method will also raise an exception if the object is not yet initialised
215:      * ({@link _IsInited()}), this happens when the object received the native container.
216:      *
217:      * This class is abstract, which means that derived classes must implement the specific
218:      * functionality of specialised data stores, for this reason this method is declared
219:      * abstract.
220:      *
221:      * @param reference            &$theObject          Object.
222:      * @param mixed                 $theIdentifier      Identifier.
223:      * @param bitfield              $theModifiers       Options.
224:      *
225:      * @access public
226:      * @return mixed                The native operation status.
227:      *
228:      * @see kFLAG_PERSIST_INSERT kFLAG_PERSIST_UPDATE kFLAG_PERSIST_MODIFY
229:      * @see kFLAG_PERSIST_REPLACE kFLAG_PERSIST_MODIFY kFLAG_PERSIST_DELETE
230:      * @see kFLAG_MODIFY_INCREMENT kFLAG_MODIFY_APPEND kFLAG_MODIFY_ADDSET
231:      * @see kFLAG_MODIFY_POP kFLAG_MODIFY_PULL kFLAG_MODIFY_MASK
232:      */
233:     abstract public function ManageObject( &$theObject,
234:                                             $theIdentifier = NULL,
235:                                             $theModifiers = kFLAG_DEFAULT );

251:     /**
252:      * <h4>Convert a binary string</h4>
253:      *
254:      * This method should be used to convert a binary string to a format compatible with the
255:      * current container.
256:      *
257:      * The first parameter represents the value to be encoded or decoded, the second boolean
258:      * parameter represents the conversion direction: <tt>TRUE</tt> means convert for
259:      * database, <tt>FALSE</tt> convert back to PHP.
260:      *
261:      * By default we convert the string to hexadecimal.
262:      *
263:      * @param mixed                 $theValue           Binary value.
264:      * @param boolean               $theSense           <tt>TRUE</tt> encode for database.
265:      *
266:      * @static
267:      * @return mixed                The encoded or decoded binary string.
268:      */
269:     static function ConvertBinary( $theValue, $theSense = TRUE )
270:     {
271:         if( $theSense )
272:             return bin2hex( $theValue );                                            // ==>
273:         
274:         return hex2bin( $theValue );                                                // ==>
275:     
276:     } // ConvertBinary.
