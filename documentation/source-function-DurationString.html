 49:     /**
 50:      * <h4>Normalise string</h4>
 51:      *
 52:      * This function can be used to format a string, the provided modifiers bitfield
 53:      * determines what manipulations are applied:
 54:      *
 55:      * <ul>
 56:      *  <li><tt>{@link kFLAG_MODIFIER_UTF8}</tt>: Convert the string to the <tt>UTF8</tt>
 57:      *      character set.
 58:      *  <li><tt>{@link kFLAG_MODIFIER_LTRIM}</tt>: Apply left trimming to the string.
 59:      *  <li><tt>{@link kFLAG_MODIFIER_RTRIM}</tt>: Apply right trimming to the string.
 60:      *  <li><tt>{@link kFLAG_MODIFIER_TRIM}</tt>: Apply both left and right trimming to the
 61:      *      string.
 62:      *  <li><tt>{@link kFLAG_MODIFIER_NULL}</tt>: If the resulting string is empty, the
 63:      *      method will return <tt>NULL</tt>.
 64:      *   <ul>
 65:      *      <li><tt>{@link kFLAG_MODIFIER_NULLSTR}</tt>: If this flag is set and the
 66:      *          resulting string is empty, the method will return the '<tt>NULL</tt>'
 67:      *          string.
 68:      *   </ul>
 69:      *  <li><tt>{@link kFLAG_MODIFIER_NOCASE}</tt>: Set the string to lowercase, this is the
 70:      *      default way to generate a case insensitive string.
 71:      *  <li><tt>{@link kFLAG_MODIFIER_URL}</tt>: URL-encode the string; note that this
 72:      *      option and {@link kFLAG_MODIFIER_HTML} are mutually exclusive.
 73:      *  <li><tt>{@link kFLAG_MODIFIER_HTML}</tt>: HTML-encode the string; note that this
 74:      *      option and {@link kFLAG_MODIFIER_URL} are mutually exclusive.
 75:      *  <li><tt>{@link kFLAG_MODIFIER_HEX}</tt>: Convert the string to hexadecimal; note
 76:      *      that this option and {@link kFLAG_MODIFIER_MASK_HASH} are mutually exclusive.
 77:      *   <ul>
 78:      *      <li><tt>{@link kFLAG_MODIFIER_HEXEXP}</tt>: Convert the string to a hexadecimal
 79:      *          expression; note that this option and {@link kFLAG_MODIFIER_MASK_HASH} are
 80:      *          mutually exclusive.
 81:      *   </ul>
 82:      *  <li><tt>{@link kFLAG_MODIFIER_HASH}</tt>: If this bit is set, the resulting string
 83:      *      will be hashed using the <tt>md5</tt> algorithm resulting in a 32 character
 84:      *      hexadecimal string; this option is mutually exclusive with the
 85:      *      {@link kFLAG_MODIFIER_MASK_HEX} option.
 86:      *   <ul>
 87:      *      <li><tt>{@link kFLAG_MODIFIER_HASH_BIN}</tt>: If this bit is set, the resulting
 88:      *          value should be a 16 character binary string; if the bit is <tt>OFF</tt>,
 89:      *          the resulting value should be a 32 character hexadecimal string.
 90:      *   </ul>
 91:      * </ul>
 92:      *
 93:      * The order in which these modifications are applied are as stated.
 94:      *
 95:      * @param string                $theString          String to normalise.
 96:      * @param bitfield              $theModifiers       Modifiers bitfield.
 97:      *
 98:      * @return mixed                Normalised string.
 99:      *
100:      * @see kFLAG_DEFAULT, kFLAG_MODIFIER_MASK
101:      * @see kFLAG_MODIFIER_UTF8
102:      * @see kFLAG_MODIFIER_LTRIM, kFLAG_MODIFIER_RTRIM, kFLAG_MODIFIER_TRIM
103:      * @see kFLAG_MODIFIER_NULL, kFLAG_MODIFIER_NULLSTR
104:      * @see kFLAG_MODIFIER_NOCASE, kFLAG_MODIFIER_URL, kFLAG_MODIFIER_HTML
105:      * @see kFLAG_MODIFIER_HEX, kFLAG_MODIFIER_HEXEXP
106:      * @see kFLAG_MODIFIER_HASH, kFLAG_MODIFIER_HASH_BIN
107:      */
108:     function StringNormalise( $theString, $theModifiers = kFLAG_DEFAULT )
109:     {
110:         //
111:         // Check if any modification was requested.
112:         //
113:         if( ($theString === NULL)                           // NULL string,
114:          || ($theModifiers === kFLAG_DEFAULT)               // or no modifiers,
115:          || (! $theModifiers & kFLAG_MODIFIER_MASK) )       // or none relevant.
116:             return $theString;                                                      // ==>
117:         
118:         //
119:         // We know now that something is to be done with the string.
120:         //
121:         
122:         //
123:         // Convert to string.
124:         //
125:         $string = (string) $theString;
126:         
127:         //
128:         // Encode string to UTF8.
129:         //
130:         if( $theModifiers & kFLAG_MODIFIER_UTF8 )
131:         {
132:             if( ! mb_check_encoding( $string, 'UTF-8' ) )
133:                 $string = mb_convert_encoding( $string, 'UTF-8' );
134:         }
135:         
136:         //
137:         // Trim.
138:         //
139:         if( $theModifiers & kFLAG_MODIFIER_MASK_TRIM )
140:         {
141:             if( ($theModifiers & kFLAG_MODIFIER_TRIM) == kFLAG_MODIFIER_TRIM ) 
142:                 $string = trim( $string );
143:             elseif( $theModifiers & kFLAG_MODIFIER_LTRIM )
144:                 $string = ltrim( $string );
145:             else
146:                 $string = rtrim( $string );
147:         }
148:         
149:         //
150:         // Handle empty string.
151:         //
152:         if( (! strlen( $string ))
153:          && ($theModifiers & kFLAG_MODIFIER_MASK_NULL) )
154:         {
155:             //
156:             // Set to NULL string.
157:             //
158:             if( ($theModifiers & kFLAG_MODIFIER_NULLSTR) == kFLAG_MODIFIER_NULLSTR )
159:                 return 'NULL';                                                      // ==>
160:             
161:             return NULL;                                                            // ==>
162:         
163:         } // Empty string and NULL mask.
164:         
165:         //
166:         // Set case insensitive.
167:         //
168:         if( $theModifiers & kFLAG_MODIFIER_NOCASE )
169:             $string = ( $theModifiers & kFLAG_MODIFIER_UTF8 )
170:                     ? mb_convert_case( $string, MB_CASE_LOWER, 'UTF-8' )
171:                     : strtolower( $string );
172:         
173:         //
174:         // URL-encode.
175:         //
176:         if( $theModifiers & kFLAG_MODIFIER_URL )
177:             $string = urlencode( $string );
178:         
179:         //
180:         // HTML-encode.
181:         //
182:         elseif( $theModifiers & kFLAG_MODIFIER_HTML )
183:             $string = htmlspecialchars( $string );
184:         
185:         //
186:         // handle HEX conversion.
187:         //
188:         if( $theModifiers & kFLAG_MODIFIER_MASK_HEX )
189:         {
190:             //
191:             // Convert to HEX string.
192:             //
193:             $string = bin2hex( $string );
194:             
195:             //
196:             // Convert to HEX expression.
197:             //
198:             if( ($theModifiers & kFLAG_MODIFIER_MASK_HEX) == kFLAG_MODIFIER_HEXEXP )
199:                 $string = "0x$string";
200:         
201:         } // HEX mask.
202:         
203:         //
204:         // Hash string.
205:         //
206:         elseif( $theModifiers & kFLAG_MODIFIER_MASK_HASH )
207:         {
208:             if( ($theModifiers & kFLAG_MODIFIER_HASH_BIN) == kFLAG_MODIFIER_HASH_BIN )
209:                 return md5( $string, TRUE );                                        // ==>
210:             
211:             return md5( $string );                                                  // ==>
212:         }
213:         
214:         return $string;                                                             // ==>
215:         
216:     } // StringNormalise.

223:     /**
224:      * <h4>Return a formatted duration</h4>
225:      *
226:      * This function will return a formatted duration string in <tt>HH:MM:SS:mmmmm</tt>
227:      * format, where <tt>H</tt> stands for hours, <tt>M</tt> stands for minutes, <tt>S</tt>
228:      * stands for seconds and <tt>m</tt> stands for milliseconds, from the value of
229:      * <tt>microtime( TRUE )</tt>.
230:      *
231:      * <i>Note: The provided value should be a difference between two timestamps taken with
232:      * <tt>microtime( TRUE )</tt>.</i>
233:      *
234:      * @param float                 $theTime            Microtime difference.
235:      *
236:      * @static
237:      * @return string               Duration string as <tt>HH:MM:SS:mmmmm</tt>.
238:      */
239:     function DurationString( $theTime )
240:     {
241:         $h = floor( $theTime / 3600.0 );
242:         $m = floor( ( $theTime - ( $h * 3600 ) ) / 60.0 );
243:         $s = floor( $theTime - ( ( $h * 3600 ) + ( $m * 60 ) ) );
244:         $l = $theTime - ( ( $h * 3600 ) + ( $m * 60 ) + $s );
245:         
246:         return sprintf( "%d:%02d:%02d:%04d", $h, $m, $s, ( $l * 10000 ) );          // ==>
247:     
248:     } // DurationString.
