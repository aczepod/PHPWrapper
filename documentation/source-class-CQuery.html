  49: /**
  50:  * <h4>Query</h4>
  51:  *
  52:  * This class implements a query.
  53:  *
  54:  * The main goal of this class is to provide a common framework and format to exchange data
  55:  * store queries or filters.
  56:  *
  57:  * The query is an array structured as follows:
  58:  *
  59:  * <ul>
  60:  *  <li><i>index</i>: The index of the root element expresses a boolean condition which
  61:  *      qualifies its content:
  62:  *   <ul>
  63:  *      <li><i>{@link kOPERATOR_AND}</i>: AND.
  64:  *      <li><i>{@link kOPERATOR_NAND}</i>: Not AND.
  65:  *      <li><i>{@link kOPERATOR_OR}</i>: OR.
  66:  *      <li><i>{@link kOPERATOR_NOR}</i>: Not OR.
  67:  *   </ul>
  68:  *      The value of this root element is an array that can be of two types:
  69:  *   <ul>
  70:  *      <li><i>Query statement</i>: A query statement defines a filter structured as
  71:  *          follows:
  72:  *       <ul>
  73:  *          <li><i>{@link kOFFSET_QUERY_SUBJECT}</i>: The subject field. It refers to the
  74:  *              object element that we are filtering.
  75:  *          <li><i>{@link kOFFSET_QUERY_OPERATOR}</i>: The filter operator. This element is
  76:  *              required and can take the following values:
  77:  *           <ul>
  78:  *              <li><i>{@link kOPERATOR_DISABLED}</i>: Disabled, it means that the filter is
  79:  *                  disabled.
  80:  *              <li><i>{@link kOPERATOR_EQUAL}</i>: Equality (<tt>=</tt>).
  81:  *              <li><i>{@link kOPERATOR_EQUAL_NOT}</i>: Inequality (<tt>!=</tt>), negates
  82:  *                  the {@link kOPERATOR_EQUAL} operator.
  83:  *              <li><i>{@link kOPERATOR_LIKE}</i>: Like, it is an accent and case
  84:  *                  insensitive equality filter.
  85:  *              <li><i>{@link kOPERATOR_LIKE_NOT}</i>: The negation of the
  86:  *                  {@link kOPERATOR_LIKE LIKE} operator.
  87:  *              <li><i>{@link kOPERATOR_PREFIX}</i>: Starts with, or prefix match.
  88:  *              <li><i>{@link kOPERATOR_CONTAINS}</i>: Contains, selects all elements that
  89:  *                  contain the match string.
  90:  *              <li><i>{@link kOPERATOR_SUFFIX}</i>: Ends with, or suffix match.
  91:  *              <li><i>{@link kOPERATOR_REGEX}</i>: Regular expression.
  92:  *              <li><i>{@link kOPERATOR_LESS}</i>: Smaller than (<tt><</tt>).
  93:  *              <li><i>{@link kOPERATOR_LESS_EQUAL}</i>: Smaller than or equal
  94:  *                  (<tt><=</tt>).
  95:  *              <li><i>{@link kOPERATOR_GREAT}</i>: Greater than (<tt>></tt>).
  96:  *              <li><i>{@link kOPERATOR_GREAT_EQUAL}</i>: Greater than or equal
  97:  *                  (<tt>>=</tt>).
  98:  *              <li><i>{@link kOPERATOR_IRANGE}</i>: Range inclusive, matches <tt>>= value
  99:  *                  <=</tt>.
 100:  *              <li><i>{@link kOPERATOR_ERANGE}</i>: Range exclusive, matches <tt>> value
 101:  *                  <</tt>.
 102:  *              <li><i>{@link kOPERATOR_NULL}</i>: Is <tt>NULL</tt> or element is missing.
 103:  *              <li><i>{@link kOPERATOR_NOT_NULL}</i>:Not <tt>NULL</tt> or element exists.
 104:  *              <li><i>{@link kOPERATOR_IN}</i>: In, or belongs to set.
 105:  *              <li><i>{@link kOPERATOR_NI}</i>: Not in, the negation of
 106:  *                  {@link kOPERATOR_IN}.
 107:  *              <li><i>{@link kOPERATOR_ALL}</i>: All, or match the full set.
 108:  *              <li><i>{@link kOPERATOR_NALL}</i>: Not all, the negation of the
 109:  *                  {@link kOPERATOR_ALL} operator.
 110:  *              <li><i>{@link kOPERATOR_EX}</i>: Expression, indicates a complex expression.
 111:  *           </ul>
 112:  *          <li><i>{@link kOFFSET_QUERY_TYPE}</i>: The data type of the
 113:  *              {@link kOFFSET_QUERY_DATA} element:
 114:  *           <ul>
 115:  *              <li><i>{@link kTYPE_STRING}</i>: String, we assume in UTF8 character set.
 116:  *              <li><i>{@link kTYPE_INT32}</i>: 32 bit signed integer.
 117:  *              <li><i>{@link kTYPE_INT64}</i>: 64 bit signed integer.
 118:  *              <li><i>{@link kTYPE_FLOAT}</i>: Floating point number.
 119:  *              <li><i>{@link kTYPE_DATE_STRING}</i>: A date.
 120:  *              <li><i>{@link kTYPE_TIME_STRING}</i>: A date and time.
 121:  *              <li><i>{@link kTYPE_STAMP}</i>: A native timestamp.
 122:  *              <li><i>{@link kTYPE_BOOLEAN}</i>: An <tt>on</tt>/<tt>off</tt> switch.
 123:  *              <li><i>{@link kTYPE_BINARY_STRING}</i>: A binary string.
 124:  *           </ul>
 125:  *          <li><i>{@link kOFFSET_QUERY_DATA}</i>: The statement test data.
 126:  *       </ul>
 127:  *      <li><i>Nested query condition</i>: A nested structure as the current one.
 128:  *   </ul>
 129:  * </ul>
 130:  *
 131:  *  @package    MyWrapper
 132:  *  @subpackage Framework
 133:  */

 207: 
 208:     /**
 209:      * <h4>Append statement</h4>
 210:      *
 211:      * This method will append the provided statement or query to the current query.
 212:      *
 213:      * The first parameter is expected to be either a single statement or a query which is
 214:      * represented by a collection of statements chained by a condition.
 215:      *
 216:      * The method expects the first parameter to be either an object derived from
 217:      * {@link CQueryStatement}, an object derived from this class or an array.
 218:      *
 219:      * If the current key of the provided statement matches one of the standard conditions
 220:      * ({@link kOPERATOR_AND}, {@link kOPERATOR_NAND}, {@link kOPERATOR_OR} or
 221:      * {@link kOPERATOR_NOR}), it is assumed that the provided statement is a <i>query</i>;
 222:      * if that is not the case, it is assumed that the provided statement is a single or
 223:      * collection of <i>statements</i>.
 224:      *
 225:      * <ul>
 226:      *  <li><i>Statement</i>: The provided statement will be appended to the current query;
 227:      *      see the {@link _HandleStatement()} method for more details.
 228:      *   <ul>
 229:      *      <li><i>Condition provided</i>:
 230:      *       <ul>
 231:      *          <li><i>Current query is empty</i>: The statement and the condition will
 232:      *              become the current query.
 233:      *          <li><i>Current query is not empty</i>: The statement will be appended to the
 234:      *              current query using the provided condition.
 235:      *       </ul>
 236:      *      <li><i>Condition not provided</i>: The condition will be determined:
 237:      *       <ul>
 238:      *          <li><i>Current query is empty</i>: The condition will default to
 239:      *              {@link kOPERATOR_AND}.
 240:      *          <li><i>Current query is not empty</i>: The current query condition will be
 241:      *              used.
 242:      *       </ul>
 243:      *   </ul>
 244:      *  <li><i>Query</i>: The provided query will be chained to the current query; see the
 245:      *      {@link _HandleCondition()} method for more details.
 246:      *   <ul>
 247:      *      <li><i>Condition provided</i>:
 248:      *       <ul>
 249:      *          <li><i>Current query is empty</i>: The provided query will become the
 250:      *              current query and the provided condition will be ignored.
 251:      *          <li><i>Current query is not empty</i>: The provided query will be chained to
 252:      *              the current query using the provided condition.
 253:      *       </ul>
 254:      *      <li><i>Condition not provided</i>:
 255:      *       <ul>
 256:      *          <li><i>Current query is empty</i>: The provided query will become the
 257:      *              current query.
 258:      *          <li><i>Current query is not empty</i>: The current query condition will be
 259:      *              used.
 260:      *       </ul>
 261:      *   </ul>
 262:      * </ul>
 263:      *
 264:      * This method makes use of two protected method for doing the dirty job:
 265:      * {@link _HandleCondition()} for queries and {@link _HandleStatement()} for statements.
 266:      *
 267:      * <i>Note: Try not to mix {@link kOPERATOR_OR} with {@link kOPERATOR_NOR} or
 268:      * {@link kOPERATOR_AND} and {@link kOPERATOR_NAND} when adding queries, the methods
 269:      * that join queries do not yet handle these combinations well: only mix them when
 270:      * adding statements.</i>
 271:      *
 272:      * @param array                 $theStatement       Statement.
 273:      * @param string                $theCondition       Statement condition.
 274:      *
 275:      * @access public
 276:      */
 277:     public function AppendStatement( $theStatement, $theCondition = NULL )
 278:     {
 279:         //
 280:         // Normalise statement.
 281:         //
 282:         if( $theStatement instanceof ArrayObject )
 283:             $theStatement = $theStatement->getArrayCopy();
 284:         
 285:         //
 286:         // Validate statement type.
 287:         //
 288:         if( ! is_array( $theStatement ) )
 289:             throw new Exception
 290:                 ( "Unsupported statement data type",
 291:                   kERROR_PARAMETER );                                           // !@! ==>
 292:         
 293:         //
 294:         // Parse by statement key.
 295:         //
 296:         reset( $theStatement );
 297:         switch( key( $theStatement ) )
 298:         {
 299:             //
 300:             // Handle query.
 301:             //
 302:             case kOPERATOR_OR:
 303:             case kOPERATOR_NOR:
 304:             case kOPERATOR_AND:
 305:             case kOPERATOR_NAND:
 306:                 
 307:                 //
 308:                 // Handle empty query.
 309:                 //
 310:                 if( ! $this->_Ready() )
 311:                     $this->exchangeArray( $theStatement );
 312:                 
 313:                 //
 314:                 // Chain query.
 315:                 //
 316:                 else
 317:                 {
 318:                     //
 319:                     // Normalise condition.
 320:                     //
 321:                     if( $theCondition === NULL )
 322:                         $theCondition = key( $this->getArrayCopy() );
 323:                     
 324:                     //
 325:                     // Handle query.
 326:                     //
 327:                     $this->_HandleCondition( $theStatement, $theCondition );
 328:                 
 329:                 } // Current query not empty.
 330:                 
 331:                 break;
 332:             
 333:             //
 334:             // Handle statement.
 335:             //
 336:             default:
 337:             
 338:                 //
 339:                 // Normalise condition.
 340:                 //
 341:                 if( $theCondition === NULL )
 342:                     $theCondition = ( $this->_Ready() )
 343:                                   ? key( $this->getArrayCopy() )
 344:                                   : kOPERATOR_AND;
 345:                 
 346:                 //
 347:                 // Handle statement.
 348:                 //
 349:                 $this->_HandleStatement( $theStatement, $theCondition );
 350:                 
 351:                 break;
 352:             
 353:         } // Parsed by statement key.
 354:     
 355:     } // AppendStatement.

 434:      *==================================================================================*/
 435: 
 436:     /**
 437:      * <h4>Handle condition</h4>
 438:      *
 439:      * This method will chain the provided query to the current query, it expects the first
 440:      * parameter to be a query array and the second parameter to be the condition that will
 441:      * be used to chain the provided query to the current query.
 442:      *
 443:      * The process of combining queries is governed by a set of three conditions: the
 444:      * <i>current</i> query condition, the <i>provided</i> condition and the
 445:      * <i>statement</i> condition. These conditions are compared as follows: 
 446:      *
 447:      * <ul>
 448:      *  <li><tt>{@link kOPERATOR_OR}</tt> is equivalent to <tt>{@link kOPERATOR_NOR}</tt>.
 449:      *  <li><tt>{@link kOPERATOR_AND}</tt> is equivalent to <tt>{@link kOPERATOR_NAND}</tt>.
 450:      * </ul>
 451:      *
 452:      * Depending on the combination of these three conditions:
 453:      *
 454:      * <ul>
 455:      *  <li><i>current</i> = <i>provided</i> = <i>statement</i>: The provided query
 456:      *      statements are recursively sent to the {@link AppendStatement()} method using
 457:      *      the condition of the provided <i>statement</i>.
 458:      *  <li><i>current</i> = <i>provided</i>: The provided query is appended to the current
 459:      *      query.
 460:      *  <li><i>current</i> = <i>statement</i>: A new empty query is created by appending
 461:      *      the old current query and the provided query using the provided condition.
 462:      *  <li><i>provided</i> = <i>statement</i>: The current query is appended to the
 463:      *      provided query.
 464:      * </ul>
 465:      *
 466:      * @param array                 $theQuery           Query.
 467:      * @param string                $theCondition       Query condition.
 468:      *
 469:      * @access protected
 470:      */
 471:     protected function _HandleCondition( $theQuery, $theCondition )
 472:     {
 473:         //
 474:         // Init local storage.
 475:         //
 476:         $provided = key( $theQuery );               // Provided query condition.
 477:         $current = key( $this->getArrayCopy() );    // Current query condition.
 478:         switch( $theCondition )
 479:         {
 480:             case kOPERATOR_OR:
 481:             case kOPERATOR_NOR:
 482:             case kOPERATOR_AND:
 483:             case kOPERATOR_NAND:
 484:                 break;
 485:             
 486:             default:
 487:                 throw new Exception
 488:                     ( "Invalid condition",
 489:                       kERROR_PARAMETER );                                       // !@! ==>
 490:         }
 491:         
 492:         //
 493:         // Serialise.
 494:         //
 495:         if( ( ( ($current == kOPERATOR_OR)
 496:              || ($current == kOPERATOR_NOR) )
 497:            && ( ($theCondition == kOPERATOR_OR)
 498:              || ($theCondition == kOPERATOR_NOR) )
 499:            && ( ($provided == kOPERATOR_OR)
 500:              || ($provided == kOPERATOR_NOR) ) )
 501:          || ( ( ($current == kOPERATOR_AND)
 502:              || ($current == kOPERATOR_NAND) )
 503:            && ( ($theCondition == kOPERATOR_AND)
 504:              || ($theCondition == kOPERATOR_NAND) )
 505:            && ( ($provided == kOPERATOR_AND)
 506:              || ($provided == kOPERATOR_NAND) ) ) )
 507:         {
 508:             //
 509:             // Use provided query.
 510:             //
 511:             foreach( $theQuery as $condition => $statements )
 512:             {
 513:                 //
 514:                 // Iterate query statements.
 515:                 //
 516:                 foreach( $statements as $statement )
 517:                     $this->AppendStatement( $statement, $condition );
 518:         
 519:             } // Used provided query.
 520:         
 521:         } // Serialise statements.
 522:         
 523:         //
 524:         // Append provided query to current query.
 525:         //
 526:         elseif( ( ( ($current == kOPERATOR_OR)
 527:                  || ($current == kOPERATOR_NOR) )
 528:                && ( ($theCondition == kOPERATOR_OR)
 529:                  || ($theCondition == kOPERATOR_NOR) ) )
 530:              || ( ( ($current == kOPERATOR_AND)
 531:                  || ($current == kOPERATOR_NAND) )
 532:                && ( ($theCondition == kOPERATOR_AND)
 533:                  || ($theCondition == kOPERATOR_NAND) ) ) )
 534:         {
 535:             //
 536:             // Get current query.
 537:             //
 538:             $tmp = $this->getArrayCopy();
 539:             
 540:             //
 541:             // Append query.
 542:             //
 543:             $tmp[ $current ][] = $theQuery;
 544:             
 545:             //
 546:             // Update current object.
 547:             //
 548:             $this->exchangeArray( $tmp );
 549:         
 550:         } // Append provided query.
 551:         
 552:         //
 553:         // Join current and provided queries with provided condition.
 554:         //
 555:         elseif( ( ( ($current == kOPERATOR_OR)
 556:                  || ($current == kOPERATOR_NOR) )
 557:                && ( ($provided == kOPERATOR_OR)
 558:                  || ($provided == kOPERATOR_NOR) ) )
 559:              || ( ( ($current == kOPERATOR_AND)
 560:                  || ($current == kOPERATOR_NAND) )
 561:                && ( ($provided == kOPERATOR_AND)
 562:                  || ($provided == kOPERATOR_NAND) ) ) )
 563:         {
 564:             //
 565:             // Get current query.
 566:             //
 567:             $tmp = $this->getArrayCopy();
 568:             
 569:             //
 570:             // Create new query.
 571:             //
 572:             $tmp = array( $theCondition => array( $tmp, $theQuery ) );
 573:             
 574:             //
 575:             // Update current object.
 576:             //
 577:             $this->exchangeArray( $tmp );
 578:         
 579:         } // Append provided to current with condition.
 580:         
 581:         //
 582:         // Append current query to provided query.
 583:         //
 584:         else
 585:         {
 586:             //
 587:             // Get current query.
 588:             //
 589:             $tmp = $this->getArrayCopy();
 590:             
 591:             //
 592:             // Append query.
 593:             //
 594:             $theQuery[ $provided ][] = $tmp;
 595:             
 596:             //
 597:             // Update current object.
 598:             //
 599:             $this->exchangeArray( $theQuery );
 600:         
 601:         } // Append current query.
 602:     
 603:     } // _HandleCondition.
 604: 
 605:      
 606:     /*===================================================================================
 607:      *  _HandleStatement                                                                *
 608:      *==================================================================================*/
 609: 
 610:     /**
 611:      * <h4>Handle statement</h4>
 612:      *
 613:      * This method will append the provided statement to the current query, it will handle
 614:      * the provided statement as a single scalar statement.
 615:      *
 616:      * @param array                 $theStatement       Statement.
 617:      * @param string                $theCondition       Statement condition.
 618:      *
 619:      * @access protected
 620:      */
 621:     protected function _HandleStatement( $theStatement, $theCondition )
 622:     {
 623:         //
 624:         // Point to provided statement.
 625:         // Here we assume it is a statement, not a nested condition.
 626:         //
 627:         if( is_int( key( $theStatement ) ) )
 628:         {
 629:             $element = current( $theStatement );
 630:             $statement = $theStatement;
 631:         }
 632:         else

 695:      * </ul>

 768:             } // Promoted statement.

 772:             //
