 29: /**
 30:  * <h4>Time-stamp</h4>
 31:  *
 32:  * This class represents a time-stamp which is equivalent to a unix timestamp, the object
 33:  * records the following information in its offsets:
 34:  *
 35:  * <ul>
 36:  *  <li><i>{@link kTAG_CUSTOM_TYPE}</i>: The constant {@link kTYPE_STAMP}.
 37:  *  <li><i>{@link kTAG_CUSTOM_DATA}</i>: The following structure:
 38:  *   <ul>
 39:  *      <li><i>{@link kTAG_STAMP_SEC}</i>: The number of seconds since midnight January 1st
 40:  *          1970 GMT, an integer.
 41:  *      <li><i>{@link kTAG_STAMP_USEC}</i>: The milliseconds part of the value, or zero.
 42:  *   </ul>
 43:  * </ul>
 44:  *
 45:  *  @package    MyWrapper
 46:  *  @subpackage Framework
 47:  */

 64:     /**
 65:      * Instantiate class.
 66:      *
 67:      * We overload the parent constructor to handle different types of data:
 68:      *
 69:      * <ul>
 70:      *  <li><i>float</i>: If the data is a float, we assume it is the result of
 71:      *      <i>microtime( TRUE )</i> in which the integral part of the number represents the
 72:      *      number of seconds since January 1st 1970 GMT, and the fractional part represents
 73:      *      the microseconds.
 74:      *  <li><i>integer</i>: If the data is an integer, we assume it represents the number of
 75:      *      seconds since January 1st 1970 GMT.
 76:      *  <li><i>MongoDate</i>: We use the object's elements.
 77:      *  <li><i>DateTime</i>: We get the date seconds count.
 78:      *  <li><i>other</i>: We assume the data is a date time string.
 79:      * </ul>
 80:      *
 81:      * No <i>NULL</i> concrete instance is allowed, all instances derived from this class
 82:      * must have a value.
 83:      *
 84:      * @param mixed                 $theData            Custom data.
 85:      *
 86:      * @access public
 87:      *
 88:      * @throws Exception
 89:      */
 90:     public function __construct( $theData = NULL )
 91:     {
 92:         //
 93:         // Get current time.
 94:         //
 95:         if( $theData === NULL )
 96:             $theData = new MongoDate();
 97:         
 98:         //
 99:         // Call parent constructor.
100:         //
101:         parent::__construct( $theData );
102:         
103:         //
104:         // Load data type.
105:         //
106:         $this->offsetSet( kTAG_CUSTOM_TYPE, kTYPE_STAMP );
107:         
108:         //
109:         // Handle float.
110:         //
111:         if( is_float( $theData ) )
112:         {
113:             $tmp = explode( '.', (string) $theData );
114:             if( count( $tmp ) > 1 )
115:             {
116:                 $usec = sprintf( '%-06s', $tmp[ 1 ] );
117:                 $this->offsetSet( kTAG_CUSTOM_DATA,
118:                                   array( kTAG_STAMP_SEC => (integer) $tmp[ 0 ],
119:                                          kTAG_STAMP_USEC => (integer) $usec ) );
120:             }
121:             else
122:                 $this->offsetSet( kTAG_CUSTOM_DATA,
123:                                   array( kTAG_STAMP_SEC => (integer) $theData,
124:                                          kTAG_STAMP_USEC => 0 ) );
125:         }
126:         
127:         //
128:         // Handle integer.
129:         //
130:         elseif( is_integer( $theData ) )
131:             $this->offsetSet( kTAG_CUSTOM_DATA,
132:                               array( kTAG_STAMP_SEC => (integer) $theData,
133:                                      kTAG_STAMP_USEC => 0 ) );
134:         
135:         //
136:         // Handle MongoDate.
137:         //
138:         elseif( $theData instanceof MongoDate )
139:             $this->offsetSet( kTAG_CUSTOM_DATA, array( kTAG_STAMP_SEC => $theData->sec,
140:                                                 kTAG_STAMP_USEC => $theData->usec ) );
141:         
142:         //
143:         // Handle DateTime.
144:         //
145:         elseif( $theData instanceof DateTime )
146:             $this->offsetSet( kTAG_CUSTOM_DATA,
147:                               array( kTAG_STAMP_SEC => $theData->format( "U" ),
148:                                      kTAG_STAMP_USEC => 0 ) );
149:         
150:         //
151:         // Handle other types.
152:         //
153:         else
154:         {
155:             $sec = strtotime( (string) $theData );
156:             if( $sec !== FALSE )
157:                 $this->offsetSet( kTAG_CUSTOM_DATA, array( kTAG_STAMP_SEC => $sec,
158:                                                     kTAG_STAMP_USEC => 0 ) );
159:             else
160:                 throw new CException( "Invalid data",
161:                                       kERROR_PARAMETER,
162:                                       kSTATUS_ERROR,
163:                                       array( 'Data' => $theData ) );            // !@! ==>
164:         }
165:     
166:     } // Constructor.
